\newcommand{\treee}[1]{
	\begin{forest}
		for tree={
			font=\sffamily\footnotesize,
			minimum height=0.75cm,
			rounded corners=4pt,
			grow'=0,
			inner ysep=8pt,
			child anchor=west,
			parent anchor=south,
			anchor=west,
			calign=first,
			edge={rounded corners},
			edge path={
				\noexpand\path [draw, \forestoption{edge}]
				(!u.south west) +(12.5pt,0) |- (.child 
				anchor)\forestoption{edge label};
			},
			before typesetting nodes={
				if n=1
				{insert before={[,phantom,minimum height=18pt]}}
				{}
			},
			fit=band,
			s sep=12pt,
			before computing xy={l=25pt},
		},#1
	\end{forest}
}
